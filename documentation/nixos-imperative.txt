============ OS Install =============

# Set root password
sudo passwd root

# SSH
ssh nixos@192.168.8.105
(In Case: ssh-keygen -R 192.168.8.105)
sudo -i

# Check Disk ID
lsblk

# Partition
cfdisk /dev/nvme1n1

# Format
mkfs.fat -F 32 -n BOOT /dev/nvme1n1p1
mkfs.btrfs -L NixOS /dev/nvme1n1p3 -f

# Create subvolume
mount /dev/nvme1n1p3 /mnt
btrfs subvolume create /mnt/@
btrfs subvolume create /mnt/@home
btrfs subvolume create /mnt/@nix
btrfs subvolume create /mnt/@log
btrfs subvolume create /mnt/@swap
umount /mnt

# Mount root
mount -o subvol=@,compress=zstd,noatime /dev/nvme1n1p3 /mnt
mkdir -p /mnt/{boot,home,nix,var/log,swap}

# Mount with optimisations
mount /dev/nvme1n1p1 /mnt/boot
mount -o subvol=@home,compress=zstd,noatime /dev/nvme1n1p3 /mnt/home
mount -o subvol=@nix,compress=zstd,noatime,lazytime /dev/nvme1n1p3 /mnt/nix
mount -o subvol=@log,compress=zstd,noatime /dev/nvme1n1p3 /mnt/var/log
mount -o subvol=@swap,noatime /dev/nvme1n1p3 /mnt/swap

# Swap File Setup
truncate -s 0 /mnt/swap/swapfile
chattr +C /mnt/swap/swapfile  # Disable CoW for swap
btrfs property set /mnt/swap/swapfile compression ""
dd if=/dev/zero of=/mnt/swap/swapfile bs=1M count=16384 status=progress # 16GB Swap
chmod 600 /mnt/swap/swapfile
mkswap /mnt/swap/swapfile
swapon /mnt/swap/swapfile #temporary

# Generate config
nixos-generate-config --root /mnt

# Create flake.nix
cd /mnt/etc/nixos

# Create and edit flake.nix
nano flake.nix

# Edit configuration.nix
nano configuration.nix

# Edit home.nix
nano home.nix

# Generate Flake Lock
nix --extra-experimental-features "nix-command flakes" flake lock

# Install nixos
nixos-install --flake .#eiri-coffee
(In case: rm -rf ~/.cache/nix)

# Password
nixos-enter --root /mnt -c 'passwd eiri'


=========== Post Install (OneTime) ============

# Download nixos-config repo to /home

# Move from /etc/nixos to ./nixos-config
sudo cp -r /etc/nixos/* ~/nixos-configs/
sudo chown -R $USER:users ~/nixos-configs/
sudo nixos-rebuild switch --flake ~/nixos-config#eiri-coffee

# Register Python Virtual Environment
cd ~/nixos-config/python-envs/kt
direnv allow
nix flake update
nix develop

# Commit
cd ~/nixos-config
git add .
nix flake update
sudo nix sudo nixos-rebuild switch --flake

# Declare password
nix-shell -p mkpasswd --run "mkpasswd -m sha-512"

# Flatpak
flatpak remote-add --if-not-exists flathub https://flathub.org/repo/flathub.flatpakrepo
flatpak update
