============ OS Install =============

# Set root password
sudo passwd root

# SSH
ssh nixos@192.168.8.105
(In Case: ssh-keygen -R 172.20.10.9)
sudo -i

# Check Disk ID
lsblk

# Partition
cfdisk /dev/nvme0n1

Device           Start        End    Sectors    Size Type
/dev/nvme0n1p1    2048    4196351    4194304      2G EFI System
/dev/nvme0n1p2 4196352    4229119      32768     16M Microsoft reserved
/dev/nvme0n1p3 4229120 1000214527  995985408  474.9G Linux filesystem

# Format
mkfs.fat -F 32 -n boot /dev/nvme0n1p1
mkfs.btrfs -L nixos /dev/nvme0n1p3 -f

# Create subvolume
mount /dev/nvme0n1p3 /mnt
btrfs subvolume create /mnt/@
btrfs subvolume create /mnt/@home
btrfs subvolume create /mnt/@nix
btrfs subvolume create /mnt/@log
btrfs subvolume create /mnt/@swap
umount /mnt

# Mount root
mount -o subvol=@,compress=zstd,noatime /dev/nvme0n1p3 /mnt
mkdir -p /mnt/{boot,home,nix,var/log,swap}

# Mount with optimisations
mount /dev/nvme0n1p1 /mnt/boot
mount -o subvol=@home,compress=zstd,noatime /dev/nvme0n1p3 /mnt/home
mount -o subvol=@nix,compress=zstd,noatime,lazytime /dev/nvme0n1p3 /mnt/nix
mount -o subvol=@log,compress=zstd,noatime /dev/nvme0n1p3 /mnt/var/log
mount -o subvol=@swap,noatime /dev/nvme0n1p3 /mnt/swap

# Swap File Setup
truncate -s 0 /mnt/swap/swapfile
chattr +C /mnt/swap/swapfile  # Disable CoW for swap
btrfs property set /mnt/swap/swapfile compression ""
dd if=/dev/zero of=/mnt/swap/swapfile bs=1M count=16384 status=progress # 16GB Swap
chmod 600 /mnt/swap/swapfile
mkswap /mnt/swap/swapfile
swapon /mnt/swap/swapfile

# Generate config
nixos-generate-config --root /mnt

# Create flake.nix
cd /mnt/etc/nixos

# Create and edit flake.nix
nano flake.nix

# Edit configuration.nix
nano configuration.nix

# Edit home.nix
nano home.nix

# Install nixos
cd
nixos-install --flake /mnt/etc/nixos#eiri-coffee
(In case: rm -rf ~/.cache/nix)

# Password
nixos-enter --root /mnt -c 'passwd eiri'


=========== Post Install (OneTime) ============

# Download nixos-config repo to /home

# Move from /etc/nixos to ./nixos-config
sudo cp -r /etc/nixos/* ~/nixos-configs/
sudo chown -R $USER:users ~/nixos-configs/
sudo nixos-rebuild switch --flake ~/nixos-config#eiri-coffee

# Register Python Virtual Environment
cd ~/nixos-config/python-envs/kt
direnv allow
nix flake update
nix develop

# Commit
cd ~/nixos-config
git add .
nix flake update
sudo nix sudo nixos-rebuild switch --flake

# Declare password
nix-shell -p mkpasswd --run "mkpasswd -m sha-512"
