# ============ Nix ============ #

# Update Repo
sudo nix-channel --update

# Install/Delete/Update
sudo nixos-rebuild switch

# Remove orphand packages
sudo nix-collect-garbag

# 删除系统Snapshot
- sudo nix-env --list-generations --profile /nix/var/nix/profiles/system
- To delete a specific one: sudo nix-env --delete-generations 10 11 --profile /nix/var/nix/profiles/system
- To delete everything older than 7 days: sudo nix-env --delete-generations 7d --profile /nix/var/nix/profiles/system
- To delete everything except the current one:
sudo nix-env --delete-generations old --profile /nix/var/nix/profiles/system
- sudo nix-collect-garbage -d

# Update boot entries
sudo /run/current-system/bin/switch-to-configuration boot
